/* 리뷰평가 */
CREATE TABLE SCOTT.REVIEW (
	RV_NO NUMBER NOT NULL, /* 리뷰번호 */
	ID VARCHAR2(20 BYTE) NOT NULL, /* 아이디 */
	AC_MST_NO NUMBER NOT NULL, /* 학원고유번호 */
	AC_REVIEWAVG NUMBER(2,1), /* 평점 */
	RV_CONTENT LONG, /* 리뷰 */
	SYMPATHY NUMBER, /* 공감 */
	REG_DATE DATE /* 리뷰작성일 */
);

COMMENT ON COLUMN SCOTT.REVIEW.AC_REVIEWAVG IS '평점';

COMMENT ON COLUMN SCOTT.REVIEW.RV_CONTENT IS '리뷰';

COMMENT ON COLUMN SCOTT.REVIEW.SYMPATHY IS '공감';

COMMENT ON COLUMN SCOTT.REVIEW.REG_DATE IS '리뷰작성일';

CREATE UNIQUE INDEX SCOTT.PK_REVIEW
